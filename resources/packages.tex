\usepackage{lmodern} % decent looking font
\usepackage{graphicx} % include image files
\usepackage[onehalfspacing]{setspace} % support for changing line spacing within the document

% document spacing (according to DHBW guidelines)
\usepackage[a4paper,margin=2.5cm,foot=1cm]{geometry} 

% paragraph spacing
\setlength{\parindent}{0mm}
\setlength{\parskip}{2mm}

% footnote spacing
\usepackage[perpage,hang]{footmisc}
\setlength{\skip\footins}{10mm} % space between main text and footnote
\setlength{\footnotesep}{5mm} % space between footnotes
\setlength{\footnotemargin}{3mm} % space after footnote number

% nicer listings
\usepackage{listings}
\lstset{
  basicstyle=\ttfamily\footnotesize,
  breaklines=true,
  frame=single}

% support links
\usepackage[hidelinks]{hyperref}
\usepackage{xurl} % clickable URLs with line breaks
